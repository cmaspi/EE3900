\section{Problem}
(Q 2.22 a)  For each of the pairs of sequences in following figure, use discrete convolution to find the response to the input $x[n]$ of the linear time-invariant system with impulse response $h[n]$
\begin{figure}[!ht]
    \centering
    \includegraphics[width=\columnwidth]{plot/fig}
    \caption{Given Plot}
    \label{magnitude}
\end{figure}
\section{Solution}
\begin{align}
    y[n]&=x[n]*h[n] \label{eq:convo}\\
    &=\displaystyle \sum_{k=-\infty}^{\infty}x[k]h[n-k]
\end{align}
From the input signal figure
\begin{align}
    x[n]=\delta[n-1] \label{eq:input}
\end{align}
The plot of $x[n]$ is given below
\begin{figure}[H]
    \centering
    \includegraphics[width=\columnwidth]{plot/x}
    \caption{plot of $x$}
    \label{x}
\end{figure}
The plot of $h[n]$ is given below 
\begin{figure}[H]
    \centering
    \includegraphics[width=\columnwidth]{plot/h}
    \caption{plot of $h$}
    \label{magnitude}
\end{figure}
from~\eqref{eq:convo} and~\eqref{eq:input}, we get
\begin{align}
    y[n]&=\delta[n-1]*h[n]\\
    &=h[n-1]
\end{align}

\begin{figure}[!ht]
    \centering
    \includegraphics[width=\columnwidth]{plot/convo}
    \caption{plot of $y$}
    \label{y}
\end{figure}
